Technical Implementation

The indicator appears dynamically based on entity type:

const recommendations = {
  'Obstacle': ['reverse', 'delete'],
  'Entity': ['collect', 'continue'],
  'Solution': ['collect'],
  'Goal': ['continue', 'collect'],
  'Location': ['continue']
};

const indicator = (action) => {
  const isRecommended = recommended.includes(action);
  if (isRecommended) {
    return `<div class="indicator"></div>`;
  }
  return '';
};
